import math

def add(num1,num2):
    return num1 + num2
def sub(num1,num2):
    return num1 - num2 
def mul(num1,num2):
    return num1 * num2
def div(num1,num2):
    return num1 / num2 
def sq(num):
    return num*num
def sqrt(num):
    return math.sqrt(num)
def cb(num):
    return num*num*num
def cbrt(num):
    return math.cbrt(num)
